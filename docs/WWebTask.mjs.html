import path from 'path'
import fs from 'fs'
import get from 'lodash-es/get.js'

/**
 * 任務調度伺服器
 *
 * @class
 * @param {Function} WOrm 輸入資料庫ORM函數
 * @param {String} url 輸入資料庫連線字串，例如w-orm-lmdb為'./db'
 * @param {String} db 輸入資料庫名稱字串
 * @param {Function} getUserByToken 輸入處理函數，函數會傳入使用者token，通過此函數處理後並回傳使用者資訊物件，並至少須提供'id'、'email'、'name'、'isAdmin'欄位，且'isAdmin'限輸入'y'或'n'
 * @param {Function} verifyClientUser 輸入驗證瀏覽使用者身份之處理函數，函數會傳入使用者資訊物件，通過此函數識別後回傳布林值，允許使用者回傳true，反之回傳false
 * @param {Function} verifyAppUser 輸入驗證應用程序使用者身份之處理函數，函數會傳入使用者資訊物件，通過此函數識別後回傳布林值，允許使用者回傳true，反之回傳false
 * @param {Object} [opt={}] 輸入設定物件，預設{}
 * @param {Integer} [opt.serverPort=11008] 輸入伺服器通訊port，預設11008
 * @returns {Object} 回傳物件，其內server為hapi伺服器實體
 */
function WWebTask(WOrm, url, db, getUserByToken, verifyClientUser, verifyAppUser, opt = {}) {
    let instWServHapiServer = null


    //check WOrm
    if (!isfun(WOrm)) {
        console.log('invalid WOrm', WOrm)
        throw new Error('invalid WOrm')
    }


    //check url
    if (!isestr(url)) {
        console.log('invalid url', url)
        throw new Error('invalid url')
    }


    //check db
    if (!isestr(db)) {
        console.log('invalid db', db)
        throw new Error('invalid db')
    }


    //check getUserByToken
    if (!isfun(getUserByToken)) {
        console.log('invalid getUserByToken', getUserByToken)
        throw new Error('invalid getUserByToken')
    }


    //check verifyClientUser
    if (!isfun(verifyClientUser)) {
        console.log('invalid verifyClientUser', verifyClientUser)
        throw new Error('invalid verifyClientUser')
    }


    //check verifyAppUser
    if (!isfun(verifyAppUser)) {
        console.log('invalid verifyAppUser', verifyAppUser)
        throw new Error('invalid verifyAppUser')
    }


    //serverPort
    let serverPort = get(opt, 'serverPort')
    if (!ispint(serverPort)) {
        serverPort = 11008
    }
    serverPort = cint(serverPort)


    //useCheckUser
    let useCheckUser = get(opt, 'useCheckUser', false)


    //getUserById
    let getUserById = get(opt, 'getUserById', null)


    //useExcludeWhenNotAdmin
    let useExcludeWhenNotAdmin = get(opt, 'useExcludeWhenNotAdmin', false)


    //webName
    let webName = get(opt, 'webName', {})


    //webDescription
    let webDescription = get(opt, 'webDescription', {})


    //webLogo
    let webLogo = get(opt, 'webLogo', '')


    //subfolder
    let subfolder = get(opt, 'subfolder', '')
    if (isestr(subfolder)) {
        if (strright(subfolder, 1) === '/') { //右邊不需要給「/」
            subfolder = strdelright(subfolder, 1)
        }
        if (strleft(subfolder, 1) !== '/') { //左邊需要給「/」
            subfolder = `/${subfolder}`
        }
    }


    //urlRedirect
    let urlRedirect = get(opt, 'urlRedirect', '')


    //params
    let showLanguage = get(opt, 'showLanguage', 'y')
    let language = get(opt, 'language', 'eng')
    let logFd = get(opt, 'logFd', './logs') //srLog 寫入之 log 資料夾


    //kpLangExt
    let kpLangExt = get(opt, 'kpLangExt', {})


    //kpLang
    let kpLang = procLang({ kpLangExt, webName, webDescription })


    //WServOrm
    let optWServOrm = {
        useCheckUser,
        getUserById,
        useExcludeWhenNotAdmin,
    }
    let wp = {}
    try {
        wp = WServOrm(ds, WOrm, url, db, optWServOrm)
    }
    catch (err) {
        //ORM 為必要依賴：初始化失敗即 fail-fast，不以 wp={} 帶 undefined woItems/procOrm 殘缺啟動
        console.log('WServOrm init failed', err)
        throw err
    }
    let { woItems, procOrm } = wp


    //srLog（後端 log；錯誤一律 log err-key 供識別）
    let srLog = srLogInit(opt)

    //maskTok：debug console.log 不印完整 token/authorization（憑證），僅前 4 碼 + ***
    let maskTok = (t) => isestr(t) ? (strleft(t, 4) + '***') : t

    //runApi: 統一 HTTP handler 收尾 — pm2resolve + isErr 還原 + 失敗記 srLog.error (與 kpFunExt 之 wrapErrLog 對稱)
    let runApi = async (event, core) => {
        let r = await pm2resolve(core)()
        if (isErr(r.msg)) {
            r.msg = r.msg.message
        }
        if (r.state !== 'success') {
            srLog.error({ event: 'api/' + event, err: r.msg })
        }
        return r
    }

    //kpFunExt 邊界統一處理：成功記 info；catch 後端錯誤 → srLog 該 err-key → 原樣 re-reject。
    //event 統一加 'kpfun-' 前綴 → 專用事件、與 api/verifyConn 區分。
    let wrapErrLog = (event, fn) => {
        let ev = 'kpfun-' + event
        return async (...args) => {
            try {
                let result = await fn(...args)
                srLog.info({ event: ev }) //成功也記 info
                return result
            }
            catch (err) {
                srLog.error({ event: ev, err })
                return Promise.reject(err)
            }
        }
    }


    //kmx: per-key in-memory mutex, 同 key 序列化、不同 key 並行. 任務生命週期 critical 路徑用.
    let kmx = pmKeyMutex()


    //procCore
    let pc = procCore({ woItems, procOrm, ds, srLog, kmx })
    let { saveFile, getFileInfo, readFile, getStats } = pc


    //封存掃描 (spec D12 方向②: 冷熱分離): 啟動後 30s 首掃 + 每小時掃一次, 將過齡終態任務/舊訊息搬至冷表.
    //timer 註冊於此 (不放 procCore) → procCore 保持無 timer, unit 測試可獨立建構與退出.
    //archiveAfterDays<=0 停用 (archiveSweep 內亦有守衛).
    let archiveAfterDays = cint(get(opt, 'archiveAfterDays', 30))
    if (archiveAfterDays > 0) {
        let lockingForArchive = false
        let runArchiveSweep = async () => {
            if (lockingForArchive) {
                return
            }
            lockingForArchive = true
            await pc.archiveSweep(archiveAfterDays)
                .then((r) => {
                    if (r.tasksMoved > 0 || r.messagesMoved > 0) {
                        srLog.info({ event: 'archiveSweep', tasksMoved: r.tasksMoved, messagesMoved: r.messagesMoved })
                    }
                })
                .catch((err) => {
                    srLog.error({ event: 'archiveSweep-error', msg: (typeof err === 'string') ? err : ((err && err.message) || 'unknown') })
                })
                .finally(() => {
                    lockingForArchive = false
                })
        }
        setTimeout(runArchiveSweep, 30 * 1000)
        setInterval(runArchiveSweep, 60 * 60 * 1000)
    }


    //getWebInfor
    let getWebInfor = (userId) => {
        return {

            // webName, //已併入kpLang
            // webDescription, //已併入kpLang
            webLogo,

            showLanguage,
            language,
            kpLang,

        }
    }


    //checkUser
    let checkUser = async(user) => {
        let id = get(user, 'id', '')
        if (!isestr(id)) {
            console.log('user', user)
            console.log('can not get the userId')
            return Promise.reject('errUserIdMissing')
        }
        let email = get(user, 'email', '')
        if (!isestr(email)) {
            console.log('user', user)
            console.log('can not get the email of user')
            return Promise.reject('errUserEmailMissing')
        }
        let name = get(user, 'name', '')
        if (!isestr(name)) {
            console.log('user', user)
            console.log('can not get userName')
            return Promise.reject('errUserNameMissing')
        }
        let isAdmin = get(user, 'isAdmin', '')
        if (isAdmin !== 'y' && isAdmin !== 'n') {
            console.log('user', user)
            console.log('user.isAdmin is not y or n', user.isAdmin)
            console.log('can not get the role of user')
            return Promise.reject('errUserRoleMissing')
        }
        return true
    }


    //getTokenUser
    let getTokenUser = async (token) => {

        //getUserByToken
        let userSelf = getUserByToken(token)
        if (ispm(userSelf)) {
            userSelf = await userSelf
        }

        //check
        if (true) {
            await checkUser(userSelf)
        }

        return userSelf
    }


    //p
    let p = {
        checkToken: async(token) => {

            //getUserByToken
            let u = await getUserByToken(token)

            //checkUser
            await checkUser(u)

            return true
        },
    }


    //getAndVerifyClientUser
    let getAndVerifyClientUser = async (token, from = '') => {

        //getTokenUser
        let user = await getTokenUser(token)

        //verifyClientUser
        let b = verifyClientUser(user, from)
        if (ispm(b)) {
            b = await b
        }

        //check
        if (!b) {
            console.log('user', user)
            console.log(`user does not have permission`)
            return Promise.reject('errUserNoPermission')
        }

        srLog.info({ event: 'verifyClientUser', from, userId: get(user, 'id', '') }) //瀏覽使用者驗證通過
        return user
    }


    //getAndVerifyAppUser
    let getAndVerifyAppUser = async (token, from = '') => {

        //user
        let user = null
        if (isestr(token)) {
            user = getUserByToken(token)
            if (ispm(user)) {
                user = await user
            }
        }

        //check
        if (!iseobj(user)) {
            console.log(`token`, maskTok(token))
            console.log(`can not find the user from token`)
            return Promise.reject('errUserNotFound')
        }

        //checkUser: 與 client 路徑對稱 (id/email/name/isAdmin 缺一即拒).
        //agent 建議以 SSO app token (isApp='y') 認證, 其解析之虛擬使用者四欄皆為非空占位值, 不受本檢查影響.
        await checkUser(user)

        //verifyAppUser
        let b = verifyAppUser(user, from)
        if (ispm(b)) {
            b = await b
        }

        //check
        if (!b) {
            console.log('user', user)
            console.log(`user does not have permission`)
            return Promise.reject('errUserNoPermission')
        }

        srLog.info({ event: 'verifyAppUser', from, userId: get(user, 'id', '') }) //應用程式使用者驗證通過
        return user
    }


    //parsePayload
    let parsePayload = async (req) => {

        //payload
        let payload = get(req, 'payload')

        //to obj
        if (isestr(payload)) {
            payload = j2o(payload)
        }

        //check
        if (!iseobj(payload)) {
            console.log('payload', payload)
            console.log(`invalid payload in req`)
            return Promise.reject('errPayloadInvalid')
        }

        return payload
    }


    //pathStaticFiles
    let pathStaticFiles = 'dist'
    let npmPathStaticFiles = './node_modules/w-web-task/dist'
    if (fsIsFolder(npmPathStaticFiles)) {
        pathStaticFiles = npmPathStaticFiles
    }


    //subfolder 注入: index.tmp → index.html
    let fnEntryIn = 'index.tmp'
    let fnEntryOut = 'index.html'
    try {
        let fpEntryIn = path.resolve(pathStaticFiles, fnEntryIn)
        if (!fsIsFile(fpEntryIn)) {
            fpEntryIn = path.resolve(pathStaticFiles, fnEntryOut) //本機開發另使用html替代tmp
        }
        if (!fsIsFile(fpEntryIn)) {
            console.log('fpEntryIn', fpEntryIn)
            throw new Error(`invalid fpEntryIn`)
        }
        let fpEntryOut = path.resolve(pathStaticFiles, fnEntryOut)
        let c = fs.readFileSync(fpEntryIn, 'utf8')
        c = replace(c, '/mtask/', '{sfd}/') //方法同genEntry
        c = replace(c, '{sfd}', subfolder)
        c = replace(c, '{urlRedirect}', urlRedirect)
        c = replace(c, '{language}', language) //注入初始語系 → window.___pmwtask___.language
        fs.writeFileSync(fpEntryOut, c, 'utf8')
    }
    catch (err) {
        console.log(err)
        console.log(`can not generate ${fnEntryOut}`)
    }


    //apis (對外 agent 面 = HTTP REST). token 由 GET ?token= 或 POST payload 帶入; getAndVerifyAppUser 守門.
    let apis = [

        {
            method: 'GET',
            path: '/api/getUserByToken', //未登入主頁時需先檢測token, getUserByToken為w-ui-loginout預設值
            handler: async function (req, res) {

                async function core() {

                    //token
                    let token = get(req, 'query.token', '')

                    //check
                    if (!isestr(token)) {
                        console.log('req.query', get(req, 'query'))
                        console.log('token', maskTok(token))
                        console.log('[API]getUserByToken/check token: invalid token')
                        return Promise.reject('errTokenNoPermission')
                    }

                    //getAndVerifyClientUser
                    let user = await getAndVerifyClientUser(token, 'getUserByToken')

                    //check
                    if (!iseobj(user)) {
                        console.log('token', maskTok(token))
                        console.log('[API]getUserByToken/check user: invalid user')
                        return Promise.reject('errTokenNoPermission')
                    }

                    srLog.info({ event: 'api/getUserByToken', userId: get(user, 'id', '') })
                    return user
                }

                return await runApi('getUserByToken', core)
            },
        },

        {
            method: 'GET',
            path: '/api/getChannels', //查頻道清單 (可依主責 agent 過濾)
            handler: async function (req, res) {

                async function core() {

                    //token
                    let token = get(req, 'query.token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'getChannels')
                    let userId = get(user, 'id', '')

                    //agentId (查詢過濾參數)
                    let agentId = get(req, 'query.agentId', '')

                    //getChannelsList
                    let r = await pc.getChannelsList(userId, agentId)

                    srLog.info({ event: 'api/getChannels', userId })
                    return r
                }

                return await runApi('getChannels', core)
            },
        },

        {
            method: 'GET',
            path: '/api/getRecentMessages', //查頻道訊息
            handler: async function (req, res) {

                async function core() {

                    //token
                    let token = get(req, 'query.token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'getRecentMessages')
                    let userId = get(user, 'id', '')

                    //params
                    let channelId = get(req, 'query.channelId', '')
                    let n = cint(get(req, 'query.n', 20))
                    let afterId = get(req, 'query.afterId', '')

                    //getMessages
                    let r = await pc.getMessages(userId, channelId, n, afterId)

                    srLog.info({ event: 'api/getRecentMessages', userId })
                    return r
                }

                return await runApi('getRecentMessages', core)
            },
        },

        {
            method: 'GET',
            path: '/api/getRecentTasks', //查頻道任務
            handler: async function (req, res) {

                async function core() {

                    //token
                    let token = get(req, 'query.token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'getRecentTasks')
                    let userId = get(user, 'id', '')

                    //params
                    let channelId = get(req, 'query.channelId', '')
                    let n = cint(get(req, 'query.n', 20))
                    let state = get(req, 'query.state', '')
                    let order = get(req, 'query.order', 'desc')

                    //getTasks
                    let r = await pc.getTasks(userId, channelId, n, state, order)

                    srLog.info({ event: 'api/getRecentTasks', userId })
                    return r
                }

                return await runApi('getRecentTasks', core)
            },
        },

        {
            method: 'GET',
            path: '/api/getArchivedTasks', //查頻道封存任務 (冷表, 稽核/回顧用; 見 spec D12)
            handler: async function (req, res) {

                async function core() {

                    //token
                    let token = get(req, 'query.token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'getArchivedTasks')
                    let userId = get(user, 'id', '')

                    //params
                    let channelId = get(req, 'query.channelId', '')
                    let n = cint(get(req, 'query.n', 20))
                    let order = get(req, 'query.order', 'desc')

                    //getArchivedTasks
                    let r = await pc.getArchivedTasks(userId, channelId, n, order)

                    srLog.info({ event: 'api/getArchivedTasks', userId })
                    return r
                }

                return await runApi('getArchivedTasks', core)
            },
        },

        {
            method: 'GET',
            path: '/api/getArchivedMessages', //查頻道封存訊息 (冷表, 稽核用; 見 spec D12)
            handler: async function (req, res) {

                async function core() {

                    //token
                    let token = get(req, 'query.token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'getArchivedMessages')
                    let userId = get(user, 'id', '')

                    //params
                    let channelId = get(req, 'query.channelId', '')
                    let n = cint(get(req, 'query.n', 20))

                    //getArchivedMessages
                    let r = await pc.getArchivedMessages(userId, channelId, n)

                    srLog.info({ event: 'api/getArchivedMessages', userId })
                    return r
                }

                return await runApi('getArchivedMessages', core)
            },
        },

        {
            method: 'GET',
            path: '/api/getTask', //查單一任務
            handler: async function (req, res) {

                async function core() {

                    //token
                    let token = get(req, 'query.token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'getTask')
                    let userId = get(user, 'id', '')

                    //params
                    let taskId = get(req, 'query.taskId', '')

                    //getTask
                    let r = await pc.getTask(userId, taskId)

                    srLog.info({ event: 'api/getTask', userId })
                    return r
                }

                return await runApi('getTask', core)
            },
        },

        {
            method: 'POST',
            path: '/api/postMessage', //傳訊至頻道 (asTask=true 同時開任務)
            handler: async function (req, res) {

                async function core() {

                    //payload (先 parse, token 可在 query 或 payload)
                    let payload = await parsePayload(req)
                    let token = get(req, 'query.token', '') || get(payload, 'token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'postMessage')
                    let userId = get(user, 'id', '')

                    let channelId = get(payload, 'channelId', '')
                    let content = get(payload, 'content', '')
                    let opt = {
                        asTask: get(payload, 'asTask', false),
                        title: get(payload, 'title', ''),
                        senderType: get(payload, 'senderType', 'human'),
                        attachments: get(payload, 'attachments', []),
                    }

                    //postMessage
                    let r = await pc.postMessage(userId, channelId, content, opt)

                    srLog.info({ event: 'api/postMessage', userId })
                    return r
                }

                return await runApi('postMessage', core)
            },
        },

        {
            method: 'POST',
            path: '/api/claimTask', //認領任務
            handler: async function (req, res) {

                async function core() {

                    //payload (先 parse, token 可在 query 或 payload)
                    let payload = await parsePayload(req)
                    let token = get(req, 'query.token', '') || get(payload, 'token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'claimTask')
                    let userId = get(user, 'id', '')

                    let taskId = get(payload, 'taskId', '')

                    //claimTask
                    let r = await pc.claimTask(userId, taskId)

                    srLog.info({ event: 'api/claimTask', userId })
                    return r
                }

                return await runApi('claimTask', core)
            },
        },

        {
            method: 'POST',
            path: '/api/respondTask', //回應任務 (唯一終態入口)
            handler: async function (req, res) {

                async function core() {

                    //payload (先 parse, token 可在 query 或 payload)
                    let payload = await parsePayload(req)
                    let token = get(req, 'query.token', '') || get(payload, 'token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'respondTask')
                    let userId = get(user, 'id', '')

                    let taskId = get(payload, 'taskId', '')
                    let result = get(payload, 'result', '')
                    let state = get(payload, 'state', 'done')
                    let errorKey = get(payload, 'errorKey', '')

                    //respondTask (不傳 senderType → 預設 'agent')
                    let r = await pc.respondTask(userId, taskId, result, state, errorKey)

                    srLog.info({ event: 'api/respondTask', userId })
                    return r
                }

                return await runApi('respondTask', core)
            },
        },

        {
            method: 'POST',
            path: '/api/resetTask', //重置任務
            handler: async function (req, res) {

                async function core() {

                    //payload (先 parse, token 可在 query 或 payload)
                    let payload = await parsePayload(req)
                    let token = get(req, 'query.token', '') || get(payload, 'token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'resetTask')
                    let userId = get(user, 'id', '')

                    let taskId = get(payload, 'taskId', '')

                    //resetTask
                    let r = await pc.resetTask(userId, taskId)

                    srLog.info({ event: 'api/resetTask', userId })
                    return r
                }

                return await runApi('resetTask', core)
            },
        },

        {
            method: 'POST',
            path: '/api/ackChannel', //推進已讀游標
            handler: async function (req, res) {

                async function core() {

                    //payload (先 parse, token 可在 query 或 payload)
                    let payload = await parsePayload(req)
                    let token = get(req, 'query.token', '') || get(payload, 'token', '')

                    //getAndVerifyAppUser
                    let user = await getAndVerifyAppUser(token, 'ackChannel')
                    let userId = get(user, 'id', '')

            handler: async function (req, res) {

                let token = get(req, 'query.token', '')
                let id = get(req, 'query.id', '')
                try {
                    let user = await getAndVerifyAppUser(token, 'getFile')
                    let r = await readFile(id)
                    srLog.info({ event: 'api/getFile', userId: get(user, 'id', '') })
                    //安全: 僅白名單點陣圖類型可用原 Content-Type 以 inline 呈現 (聊天 <img> 需求);
                    //其餘 (含 text/html, image/svg+xml 等可挾帶 script 之類型) 一律強制 application/octet-stream + attachment 下載,
                    //避免上傳惡意 HTML/SVG 後經 getFile 於本站原點 inline 渲染造成 XSS. 並加 nosniff 禁瀏覽器 MIME 嗅探.
                    let safeInlineTypes = ['image/png', 'image/jpeg', 'image/jpg', 'image/gif', 'image/webp', 'image/bmp', 'image/avif']
                    let fileType = r.info.type || ''
                    let inline = safeInlineTypes.includes(fileType)
                    let fname = encodeURIComponent(r.info.name || id)
                    return res.response(r.buf)
                        .type(inline ? fileType : 'application/octet-stream')
                        .header('Content-Disposition', `${inline ? 'inline' : 'attachment'}; filename="${fname}"`)
                        .header('X-Content-Type-Options', 'nosniff')
                }
                catch (err) {
                    let key = isestr(err) ? err : get(err, 'message', 'errFileNotFound')
                    srLog.error({ event: 'api/getFile', err: key })
                    return res.response(JSON.stringify({ state: 'error', msg: key })).code(404).type('application/json')
                }
            },
